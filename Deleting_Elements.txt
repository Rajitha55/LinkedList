//Program to delete a node from Linked list

#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node* next;
    Node(int val)
    {
        data=val;
        next=NULL;
    }
};

void deletebegin(Node *&head)
{
    if(head==NULL)
       return;
    
    Node *temp=head;
    head=head->next;
    delete temp;
}

void deleteend(Node *&head)
{
    if(head==NULL)
       return;
    Node *temp=head;
    if(head->next == NULL)
    {
        temp=head;
        head=NULL;
        delete temp;
        return;
    }
    
    while(temp->next->next != NULL)
        temp=temp->next;
    delete temp->next;
    temp->next=NULL;
}

void deleteval(Node *&head,int val)
{
    if(head==NULL)
       return;
       
    if(head->data==val)
    {
       deletebegin(head);
       return;
    } 
    
    Node *temp=head;
    while(temp->next!=NULL && temp->next->data!=val)
       temp=temp->next;
    
    if(temp->next==NULL)
       return;
       
    Node *temp1=temp->next;
    temp->next=temp->next->next;
    delete temp1;
}

int main()
{
    Node *head=NULL;
    head=new Node(10);
    head->next=new Node(20);
    head->next->next=new Node(30);
    //head->next->next->next=new Node(40);
    
    Node *temp=head;
    while(temp!=NULL)
    {
        cout<<temp->data<<" ";
        temp=temp->next;
    }
    cout<<endl;
    deletebegin(head);
    deleteend(head);
    deleteval(head,20);
    
    temp=head;
    if(temp==NULL)
        cout<<"null";
    
    else
    {
    while(temp!=NULL)
    {
        cout<<temp->data<<" ";
        temp=temp->next;
    }
    }
    return 0;
}
